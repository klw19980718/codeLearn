flutter中的div---container 容器组件

options名称value值类型含义
alignmentAlignment.cetner内容对齐方式
paddingnew EdgeInsets.fromLTRB(0,0,0,0)css中得padding
marginnew EdgeInsets.fromLTRB(0,0,0,0)css中得margin
width100.0宽度
height100.0高度
decorationnew BoxDecoration(color,border,borderRadius,boxShadow)装饰(颜色阴影边框圆角等)
decoration.colorColors背景颜色
decoration.bordernew Border.all(color,width:5.0)边框
decoration.borderRadiusnew BorderRadius.circular(5)圆角
decoration.boxShadownew BoxShadow(color,blurRadius:5.0)阴影
decoration.imagenew DecorationImage(image: new NetworkImage(url),fit,repeat,...)背景图片
childnew Text() / container() / ...子组件
    
import 'package:flutter/material.dart';

void main() {
  runApp(new MaterialApp(
    home: new Scaffold(
        appBar: new AppBar(
          title: new Text('hello world'),
        ),
        body: new Column(children: [
          new MyApp(),
        ])),
  ));
}

class MyApp extends StatelessWidget {
  MyApp({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return new Center(
      child: new Container(
        width: 150.0, // 宽度
        height: 150.0, // 高度
        margin: new EdgeInsets.fromLTRB(50, 250, 50, 50),
        padding: new EdgeInsets.all(10),
        decoration: new BoxDecoration(
          color: Colors.red[300],
          border: new Border.all(
            color: Colors.yellow,
            width: 2.0,
          ),
          borderRadius: new BorderRadius.circular(5),
          boxShadow: [
            new BoxShadow(
              color: Colors.black,
              blurRadius: 5.0,
            )
          ],
          gradient: new LinearGradient(colors: [Colors.red, Colors.yellow]),
        ),
        child: new Text(
          "你好你123",
          style: new TextStyle(
            color: Colors.white,
            fontSize: 25.0,
          ),
        ),
        alignment: Alignment.topRight, // 容器内元素的方位
      ),
    );
  }
}

flutter中text---text 组件

options名称value值类型含义
stylenew TextStyle(color,fontSize,fontWeight,fontStyle)字体样式
textAlignTextAlign.center对齐
overflowTextOverflow.ellipsis隐藏方式

class MyText extends StatelessWidget {
  const MyText({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return new Text(
      '我是文本',
      style: new TextStyle(
        color: Colors.red,
        fontSize: 10,
        fontStyle:FontStyle.italic,
        fontWeight: FontWeight.w100,
      ),
  
      textAlign: TextAlign.center,
      overflow: TextOverflow.ellipsis,
      maxLines: 2,

    );
  }
}

flutter中得image组件

options名称value值类型含义
widthnumber宽度
heightnumber高度
fitBoxFit.contain显示样式
alignmentAlignment.bottomLeft对齐方式
repeatImageRepeat.repeat,是否重复覆盖
class MyButton extends StatelessWidget {
  const MyButton({Key? key}) : super(key: key);

  @override
  Widget build(BuildContext context) {
    return Container(
        width: 150,
        height: 150,
        decoration: new BoxDecoration(
          color: Colors.red,
        ),
        child: new Image.network(
          'https://img.alicdn.com/imgextra/i3/2201168165444/O1CN01zkMmUL1q5P7tZwOCr_!!2201168165444.jpg',
          width: 0,
          height: 0,
          fit:BoxFit.contain,
          alignment:Alignment.bottomLeft,
          repeat:ImageRepeat.repeat,

        ));
  }
}
